@startuml


Class Usuario {
- nome: String
- senha: String
- tipo: TipoUsuario
- tarefas_habilitadas: List<Tarefa>
}

Enum TipoUsuario {
ADMIN
FUNCIONARIO
}

Class Tarefa {
- tempos: List<Tempo>
- titulo: String
- descricao: String
- comentario: String
- data_entrega: LocalDateTime
- tipo_atividade: TipoAtividade
- estado: Estado
- dificuldade: Dificuldade
}

Enum Dificuldade {
FACIL
MEDIO
DIFICIL
}

Enum Estado {
PENDENTE
EM_ATIVIDADE
CONCLUIDO
NA_LIXEIRA
}

Class TipoAtividade {
- nome: String
- descricao: String
}

Class GerenciadorTarefas {
+ alterar_contagem(tarefa: Tarefa):Tempo
+ mover_tarefa(tarefa: Tarefa, ListaTarefa: tarefas): void
+ criar_tarefa(titulo: String, descricao: String, data_entrega: LocalDateTime, tipo_atividade: TipoAtividade, dificuldade: Dificuldade): Tarefa
+ modificar_tarefa(titulo: String, descricao: String, comentario: String, data_entrega: LocalDateTime, tipo_atividade: TipoAtividade, estado: Estado, dificuldade: Dificuldade): Tarefa
}

Class GerenciadorUsuario {
+ criar_usuario(nome: String, senha: String, tipo: TipoUsuario):Usuario
+ modificar_usuario(usuario, String, nome: String, senha: String, tipo, TipoUsuario): void
+ deletar_usuario(): void
}

Class Tempo {
- data_inicial: LocalDateTime
- data_final: LocalDateTime

+ terminar_tempo(): void
}


Class Calendario {
- dias: ListaTarefa[mes]

+ coletar_tarefas(mes: int): void
}


Tarefa *-- TipoAtividade
Tarefa *-- Dificuldade
Tarefa *-- Estado
Usuario *-- TipoUsuario

GerenciadorTarefas <-- Tarefa
GerenciadorTarefas <-- Dificuldade
GerenciadorTarefas <-- Estado
GerenciadorTarefas <-- TipoAtividade
GerenciadorTarefas <-- Tempo

GerenciadorUsuario <-- Usuario
GerenciadorUsuario <-- TipoUsuario


Tarefa *-- Tempo
Usuario o-- Tarefa

@enduml
